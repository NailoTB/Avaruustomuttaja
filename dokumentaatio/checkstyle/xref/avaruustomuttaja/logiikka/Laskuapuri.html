6   package avaruustomuttaja.logiikka;
7   
8   import avaruustomuttaja.kappale.Kappale;
9   
10  public class Laskuapuri {
11  
12      private double g = 6.674;
13      /** 
14       * Laskuapurin konstruktori.
15       * 
16       * Täällä ei tapahdu mitään
17       */
18      public Laskuapuri() {
19  
20      }
21  
22      /**
23       * Metodi laskee kahden kappaleen välisen x-suuntaisen etäisyyden.
24       *
25       * @param vetaja Ensimmäinen kappale.
26       * @param kappale Toinen kappale.
27       * @return kappaleiden x-suuntainen etäisyys.
28       */
29      public double xPaikkaMuutoksenLaskija(Kappale vetaja, Kappale kappale) {
30          double deltapaikkaX = Math.abs(vetaja.getPaikkaX() - kappale.getPaikkaX());
31          return deltapaikkaX;
32      }
33  
34      /**
35       * Metodi laskee kahden kappaleen välisen y-suuntaisen etäisyyden.
36       *
37       * @param vetaja Ensimmäinen kappale.
38       * @param kappale Toinen kappale.
39       * @return kappaleiden y-suuntainen etäisyys.
40       */
41      public double yPaikkaMuutoksenLaskija(Kappale vetaja, Kappale kappale) {
42          double deltapaikkaY = Math.abs(vetaja.getPaikkaY() - kappale.getPaikkaY());
43          return deltapaikkaY;
44      }
45  
46      /**
47       * Metodi laskee kahden kappaleen välisen kulman trigonometrisesti.
48       *
49       * @param deltaPaikkaX Kappaleiden välinen x-suuntainen etäisyys.
50       * @param deltaPaikkaY Kappaleiden välinen y-suuntainen etäisyys.
51       * @return kappaleiden välinen kulma
52       */
53      public double kulmanLaskija(double deltaPaikkaX, double deltaPaikkaY) {
54          if (deltaPaikkaY == 0) {
55              return Math.PI / 2;
56          }
57          double kulma = Math.atan(deltaPaikkaX / deltaPaikkaY);
58          return kulma;
59      }
60  
61      /**
62       * Metodi laskee kahden kappaleen välisen voiman Newtonin painovoimalaista.
63       *
64       * @param kappale Kappale, johon painovoima lasketaan.
65       * @param vetaja Kappale, joka "vetää" toista kappaletta.
66       * @param etaisyys Kappaleiden välinen etäisyys.
67       * @return vetajan aiheuttaman voiman suuruus kappaleeseen
68       */
69      public double voimanLaskija(Kappale kappale, Kappale vetaja, double etaisyys) {
70          double voima = g * vetaja.getMassa() / Math.pow(etaisyys, 2);
71          return voima;
72      }
73  
74      /**
75       * Metodi laskee kahden kappaleen etäisyyden Pythagoraan lauseesta.
76       *
77       * @param deltaPaikkaX Kahden kappaleen x-suuntainen etäisyys.
78       * @param deltaPaikkaY Kahden kappaleen y-suuntainen etäisyys.
79       * @return kahden kappaleen etäisyys
80       */
81      public double etaisyydenLaskija(double deltaPaikkaX, double deltaPaikkaY) {
82          double etaisyys = Math.sqrt(Math.pow(deltaPaikkaX, 2) + Math.pow(deltaPaikkaY, 2));
83          return etaisyys;
84      }
85  
86      /**
87       * Metodi laskee voiman X-komponentin.
88       * <p>
89       * Metodi laskee X-komponentin suuruuden trigonometrisesti
90       *
91       * @param voima Kokonaisvoiman suuruus
92       * @param kulma Kahden kappaleen välinen kulma
93       * @return voiman X-komponentin suuruus
94       */
95      public double voimanXKomponentinLaskija(double voima, double kulma) {
96          double voimaX = Math.sqrt(Math.pow(voima, 2) - Math.pow(voima * Math.cos(kulma), 2));
97          return voimaX;
98      }
99  
100     /**
101      * Metodi laskee voiman Y-komponentin.
102      * <p>
103      * Metodi laskee Y-komponentin suuruuden trigonometrisesti
104      *
105      * @param voima Kokonaisvoiman suuruus
106      * @param kulma Kahden kappaleen välinen kulma
107      * @return voiman Y-komponentin suuruus
108      */
109     public double voimanYKomponentinLaskija(double voima, double kulma) {
110         double voimaY = Math.sqrt(Math.pow(voima, 2) - Math.pow(voima * Math.sin(kulma), 2));
111         return voimaY;
112     }
113 }
